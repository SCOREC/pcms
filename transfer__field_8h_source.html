80 void transfer_field(const SourceField& source, TargetField& target,
81  FieldTransferMethod transfer_method,
82  FieldEvaluationMethod evaluation_method)
83 {
84  PCMS_FUNCTION_TIMER;
85  switch (transfer_method) {
86  case FieldTransferMethod::None: return;
87  case FieldTransferMethod::Copy:
88  if constexpr(std::is_same_v<SourceField, TargetField>) {
89  copy_field(source, target);
90  }
91  else {
92  std::cerr<<"Source field and destination field must have same type to copy!\n";
93  std::abort();
94  }
95  return;
96  case FieldTransferMethod::Interpolate:
97  switch (evaluation_method) {
98  case FieldEvaluationMethod::None:
99  std::cerr << "Cannot interpolate field with no evaluation method!";
100  std::terminate();
101  case FieldEvaluationMethod::Lagrange1:
102  interpolate_field(source, target, Lagrange<1>{});
103  break;
104  case FieldEvaluationMethod::NearestNeighbor:
105  interpolate_field(source, target, NearestNeighbor{});
106  break;
107  // no default case for compiler error on missing cases
108  }
109  return;
110  // no default case for compiler error on missing transfer method

95 [[nodiscard]] ReverseClassificationVertex ConstructRCFromOmegaHMesh(

97 {
98  // transfer vtx classification to host
99  auto classIds_h =
100  Omega_h::HostRead<Omega_h::ClassId>(mesh.get_array<Omega_h::ClassId>(0, "class_id"));
101  auto classDims_h =
102  Omega_h::HostRead<Omega_h::I8>(mesh.get_array<Omega_h::I8>(0, "class_dim"));
103  auto vertid = Omega_h::HostRead<T>(mesh.get_array<T>(0, numbering));
104  pcms::ReverseClassificationVertex rc;
105  PCMS_ALWAYS_ASSERT(classDims_h.size() == classIds_h.size());
106  for (int i = 0; i < classDims_h.size(); ++i) {
107  pcms::DimID geom{classDims_h[i], classIds_h[i]};
108  if(index_base == IndexBase::Zero) {
109  rc.Insert(geom, vertid[i]);
110  } else {
111  PCMS_ALWAYS_ASSERT(vertid[i] > 0);
112  rc.Insert(geom, vertid[i] - 1);
113  }
114  }
115  return rc;

146  for (const auto& geom : reverse_classification_) {
147  // if the geometry is in specified overlap region
148  if (in_overlap_(geom.first.dim, geom.first.id)) {
149 
150  auto dr = std::visit(detail::GetRank{geom.first}, partition);
151  auto [it, inserted] = reverse_partition.try_emplace(dr);
152  // the map gives the local iteration order of the global ids
153  auto map = mask_.GetMap();
154  std::transform(geom.second.begin(), geom.second.end(),
155  std::back_inserter(it->second), [&map](auto v) {
156  auto idx = map[v];
157  PCMS_ALWAYS_ASSERT(idx > 0);
158  return idx - 1;
159  });
160  }
161  }

166  // in XGC the local iteration order maps directly to the global ids

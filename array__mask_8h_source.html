64  auto index_mask_view = make_array_view(index_mask);
65  Kokkos::parallel_scan(policy,detail::ComputeMaskAV<MemorySpace>{index_mask_view, mask},
66  num_active_entries_);
68  Kokkos::parallel_for(policy, detail::ScaleAV<MemorySpace>{index_mask_view, mask});
69  // does a shallow copy
